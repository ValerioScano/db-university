Metto il COUNT con le sue parentesi per verificare che il numero coincida con il risultato atteso

***1***

SELECT COUNT(*)
FROM students 
WHERE YEAR(date_of_birth) = 1990;

GROUP BY

***1***

SELECT YEAR(enrolment_date) AS year,
       COUNT(id) AS enrolments
FROM students
GROUP BY YEAR(enrolment_date)

***2***

SELECT COUNT(id), office_address
FROM teachers
GROUP BY office_address

***3***

SELECT AVG(vote), exam_id
FROM exam_student
GROUP BY exam_id

***4***

SELECT count(id) AS "numero corsi di laurea", department_id
FROM degrees
GROUP BY department_id
